Docker Compose

# 克隆仓库
git clone  && cd paperphone-plus

# 复制并编辑环境变量
cp server/.env.example server/.env
# 按需编辑：DB_PASS / JWT_SECRET / CF_CALLS_APP_ID 等

# 一键启动（含前端、后端、MySQL、Redis）
docker compose up -d
